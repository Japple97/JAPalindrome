                    //Determine if string is palindrome
                    function checkForPalindrome(inputString) {
                    inputString = inputString.toLowerCase();
                    
                    const regex = /[^a-z0-9]/gi;
                    
                    inputString = inputString.replace(regex, "");
                    
                    let reversedString = reverseString(inputString);
                    
                    
                    return inputString == reversedString;
                    
                    }
                    
                    
                    function reverseString(inputString) {
                    let reversedString = '';
                    
                    for (let index = inputString.length - 1; index >= 0; index--) {
                    reversedString += inputString[index];
                    }
                    return reversedString
                    }
                    
                    
                    //Display message to user to show
                    //if string is Palindrome or not
                    function displayResults(results, inputString) {
                    let alert = document.getElementById('alert');
                    
                    alert.classList.remove('alert-danger');
                    alert.classList.remove('alert-success');
                    
                    let messageElement = document.getElementById('message');
                    
                    if (results == true) {
                    alert.classList.add('alert-success');
                    messageElement.textContent = "Hooray! A Palindrome!";
                    } else {
                    alert.classList.add('alert-danger');
                    messageElement.textContent = "Boooo not a palindrome.";
                    }
                    
                    let resultsElement = document.getElementById('results');
                    let reversedString = reverseString(inputString);
                    resultsElement.textContent = reversedString;
                    
                    alert.classList.remove('invisible');
                    }

The code is a JavaScript code that checks if a given string is a palindrome or not. It consists of four functions.

getValues(): This is the main function that is triggered by an event, such as a button click. It retrieves the value entered by the user in an HTML input element with the ID 'userString' and stores it in the variable 'inputString'. It then passes this value to the 'checkForPalindrome' function to check if it is a palindrome or not. The result of this function is passed on to the 'displayResults' function to display the result to the user.

checkForPalindrome(inputString): This function checks if the input string is a palindrome. It starts by converting the input string to lowercase and removes any non-alphanumeric characters using a regular expression (regex). The input string is then reversed using the 'reverseString' function, and the reversed string is compared to the original string. If they are equal, the function returns true, meaning the input string is a palindrome. If they are not equal, the function returns false, meaning the input string is not a palindrome.

reverseString(inputString): This function reverses the input string. It creates an empty string 'reversedString' and loops through the input string from its last character to its first character, adding each character to the 'reversedString'. Finally, it returns the 'reversedString'.

displayResults(results, inputString): This function displays the result of the palindrome check to the user. It starts by removing any previous classes from the 'alert' element and adds either the 'alert-success' or 'alert-danger' class, depending on the result of the palindrome check. It then sets the text content of the 'message' element to display either "Hooray! A Palindrome!" if the input string is a palindrome, or "Boooo not a palindrome." if the input string is not a palindrome. Finally, it sets the text content of the 'results' element to display the reversed string and makes the 'alert' element visible by removing the 'invisible' class.
